    <!-- JavaScript Logic -->
    <script>
        // State Management
        let schedules = JSON.parse(localStorage.getItem('khasilaSchedules')) || [];
        let currentFilter = 'all';

        // DOM Elements
        const form = document.getElementById('scheduleForm');
        const listContainer = document.getElementById('scheduleList');
        const emptyState = document.getElementById('emptyState');
        const filterBtns = document.querySelectorAll('.filter-btn');

        // Helpers for categories
        const categoryConfig = {
            rutin: { icon: 'ph-calendar-check', label: 'Rutinitas', colorClass: 'category-rutin' },
            hobi: { icon: 'ph-palette', label: 'Hobi', colorClass: 'category-hobi' },
            spesial: { icon: 'ph-sparkle', label: 'Spesial', colorClass: 'category-spesial' }
        };

        // Initialize App
        function init() {
            renderSchedules();
            
            // Form Submit Event
            form.addEventListener('submit', function(e) {
                e.preventDefault();
                addSchedule();
            });

            // Filter Events
            filterBtns.forEach(btn => {
                btn.addEventListener('click', (e) => {
                    // Update active styling
                    filterBtns.forEach(b => {
                        b.classList.remove('bg-khasila-text', 'text-white');
                        b.classList.add('bg-white', 'text-gray-500');
                    });
                    e.target.classList.remove('bg-white', 'text-gray-500');
                    e.target.classList.add('bg-khasila-text', 'text-white');

                    // Set filter and render
                    currentFilter = e.target.dataset.filter;
                    renderSchedules();
                });
            });
        }

        // Add New Schedule
        function addSchedule() {
            const time = document.getElementById('timeInput').value;
            const category = document.getElementById('categoryInput').value;
            const title = document.getElementById('titleInput').value;

            const newSchedule = {
                id: Date.now().toString(),
                time: time,
                category: category,
                title: title,
                isDone: false
            };

            schedules.push(newSchedule);
            saveData();
            renderSchedules();
            form.reset();
            
            // Set default time to slightly after now for convenience
            document.getElementById('timeInput').focus();
        }

        // Toggle Done Status
        function toggleDone(id) {
            const index = schedules.findIndex(s => s.id === id);
            if (index !== -1) {
                schedules[index].isDone = !schedules[index].isDone;
                saveData();
                renderSchedules();
            }
        }

        // Delete Schedule
        function deleteSchedule(id) {
            if(confirm('Hapus kegiatan ini dari jadwal?')) {
                schedules = schedules.filter(s => s.id !== id);
                saveData();
                renderSchedules();
            }
        }

        // Save to LocalStorage
        function saveData() {
            localStorage.setItem('khasilaSchedules', JSON.stringify(schedules));
        }

        // Render List
        function renderSchedules() {
            listContainer.innerHTML = '';

            // Filter
            let filteredSchedules = schedules;
            if (currentFilter !== 'all') {
                filteredSchedules = schedules.filter(s => s.category === currentFilter);
            }

            // Sort by time
            filteredSchedules.sort((a, b) => a.time.localeCompare(b.time));

            if (filteredSchedules.length === 0) {
                emptyState.classList.remove('hidden');
                emptyState.classList.add('flex');
            } else {
                emptyState.classList.add('hidden');
                emptyState.classList.remove('flex');

                filteredSchedules.forEach((schedule, index) => {
                    const cat = categoryConfig[schedule.category];
                    const doneClass = schedule.isDone ? 'item-done' : 'bg-white';
                    const checkIcon = schedule.isDone ? 'ph-check-circle ph-fill text-green-500' : 'ph-circle text-gray-300';
                    
                    // Delay animation slightly for each item
                    const animationDelay = index * 0.1;

                    const itemHTML = `
                        <div class="schedule-item ${doneClass} p-4 rounded-2xl shadow-sm border border-gray-100 flex items-center justify-between gap-3" style="animation-delay: ${animationDelay}s">
                            
                            <!-- Left side: Time & Checkbox -->
                            <div class="flex items-center gap-4 flex-1 cursor-pointer" onclick="toggleDone('${schedule.id}')">
                                <i class="${checkIcon} text-2xl hover:scale-110 transition-transform"></i>
                                <div>
                                    <div class="flex items-center gap-2 mb-1">
                                        <span class="text-sm font-bold text-khasila-text bg-gray-100 px-2 py-0.5 rounded-md">${schedule.time}</span>
                                        <span class="text-[10px] font-bold px-2 py-0.5 rounded-full ${cat.colorClass} flex items-center gap-1">
                                            <i class="${cat.icon}"></i> ${cat.label}
                                        </span>
                                    </div>
                                    <h3 class="title-text font-bold text-[15px]">${schedule.title}</h3>
                                </div>
                            </div>

                            <!-- Right side: Actions -->
                            <button onclick="deleteSchedule('${schedule.id}')" class="text-gray-400 hover:text-red-500 transition-colors p-2 bg-white rounded-full hover:bg-red-50" title="Hapus">
                                <i class="ph-bold ph-trash"></i>
                            </button>

                        </div>
                    `;
                    listContainer.insertAdjacentHTML('beforeend', itemHTML);
                });
            }
        }

        // Run app
        init();

    </script>
